import random
f=open('weapon2.txt','w',encoding='utf-8')
f.write('名稱 傷害 冷卻 魔力消耗 速度 武器高度 武器x 攻擊高度 攻擊x 技能點數 升級技能: d傷害+1 c冷卻-0.1 m魔力消耗-1 s速度+1\n')
for i in range(10):
    basic=int(150*(1.2**(i+1)))
    f.write(f'{i+1} {basic} 0.3 0 200 35 -15 20 50 300')
    for j in range(9):
        f.write(' d*'+str(int(basic*0.3)))
        basic*=1.2*(1+random.random()*0.1)
    f.write('\n')
